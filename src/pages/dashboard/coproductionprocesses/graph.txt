
  import { Canvas } from 'reaflow';

const getSchema = () => {
    if (process && process.phases) {
      const nodes = []
      const edges = []
      process.phases.forEach(phase => {
        nodes.push(
          {
            id: phase.id,
            text: phase.phase.name
          }
        ) 
        phase.objectives.forEach(objective => {
          nodes.push(
            {
              id: objective.id,
              text: objective.objective.name
            }
          )
          edges.push(
            {
              id: `${phase.id}-${objective.id}`,
              from: phase.id,
              to: objective.id
            },
          ) 
        })
      })
      return {
        nodes: nodes,
        edges: edges
      }
    }
    return {
      nodes: [],
      edges: []
    }

  }

  const {nodes, edges} = getSchema()
  <Canvas
                pannable={false}
                direction="RIGHT"
                nodes={nodes}
                edges={edges}
              />
